It is much better than MATLAB's imagesc for visualizing depth\r\n\t\t\t\t\t\t  data. \r\n<pre class='brush: cpp'>\r\nvoid imageeq(const char* winName, cv::Mat_< float > im)\r\n{\r\n    // compute the order statistics\r\n    vector< float > values;\r\n    for(int rIter = 0; rIter < im.rows; rIter++)\r\n        for(int cIter = 0; cIter < im.cols; cIter++)\r\n                values.push_back(im.at< float >(rIter,cIter));\r\n    std::sort(values.begin(),values.end());\r\n    auto newEnd = std::unique(values.begin(),values.end());\r\n    values.erase(newEnd,values.end());\r\n    \r\n    // compute an equalized image\r\n    Mat showMe(im.rows,im.cols,DataType< uchar >::type);\r\n    float oldQuant = 0;\r\n    for(int qIter = 1; qIter <= 256; qIter++)\r\n    {\r\n        float quantile = static_cast< float >(qIter)/256;\r\n      \r\n        float thresh_low = values[oldQuant*(values.size() - 1)];\r\n        float thresh_high = values[quantile*(values.size() - 1)];\r\n        //printf(\"q = %f low = %f high = %f\\n\",quantile,thresh_low,thresh_high);\r\n        for(int rIter = 0; rIter < im.rows; rIter++)\r\n            for(int cIter = 0; cIter < im.cols; cIter++)\r\n            {\r\n                float curValue = im.at< float >(rIter,cIter);\r\n                if(curValue <= thresh_high && curValue >= thresh_low)\r\n                    showMe.at< uchar >(rIter,cIter) = qIter-1;\r\n            }\r\n        \r\n        oldQuant = quantile;\r\n    }\r\n    \r\n    imshow(winName,showMe);\r\n}\r\n</pre>\r\n\t\t\t\t\t\t</div>\r\n\t\t\t\t\t</div>\r\n\r\n\t\t\t\t\t<div class=\"post\">\r\n\t\t\t\t\t\t<h2 class=\"title\"><a href=\"#\"> Welcome to my academic homepage </a></h2>\r\n\t\t\t\t\t\t<td valign=\"center\" align = \"right\"> <img src=\"JSS3_IHOG.jpe\" alt=\"headshot\"> </td>\r\n\t\t\t\t\t\t<p class=\"meta\"><span class=\"date\">Dec 18th, 2012</span><span class=\"posted\">Posted by <a href=\"#\">James</a></span></p>\r\n\t\t\t\t\t\t<div style=\"clear: both;\">&nbsp;</div>\r\n\t\t\t\t\t\t<div class=\"entry\">\r\n\t\t\t\t\t\t  The above figure contains the <a href=\"http://mit.edu/vondrick/ihog/\"> inverse HOG visualization </a> of the HOG features in a picture of me.
